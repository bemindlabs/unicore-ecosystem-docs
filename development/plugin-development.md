# Plugin Development (Geek Edition)

This guide covers building custom plugins for the UniCore Geek Edition. Plugins extend the CLI, TUI, and REPL with custom commands, screens, and SDK bindings.

## Overview

The Geek Edition exposes a stable plugin API via the `@unicore-geek/plugin-sdk` package. Plugins are distributed as npm packages and installed via the `unicore plugin install` command.

## Prerequisites

- Node.js 20+
- pnpm 10.30+ or npm
- `@unicore-geek/plugin-sdk` (install from npm or local registry)
- A running UniCore Geek Edition instance for testing

## Plugin Structure

```
my-unicore-plugin/
├── package.json              ← npm manifest + plugin metadata
├── tsconfig.json             ← TypeScript config
├── src/
│   ├── index.ts              ← Plugin entry point (exports default class)
│   ├── commands/
│   │   └── index.ts          ← CLI command registrations
│   ├── screens/
│   │   └── index.tsx         ← TUI screen components (Ink/React)
│   └── repl/
│       └── bindings.ts       ← REPL global bindings
├── dist/                     ← Compiled output (generated by tsc)
└── README.md
```

## Plugin Manifest

The `package.json` doubles as the plugin manifest via the `unicorePlugin` key:

```json
{
  "name": "@my-org/unicore-plugin-crm-exporter",
  "version": "1.0.0",
  "description": "Export CRM contacts to CSV and Google Sheets",
  "main": "dist/index.js",
  "types": "dist/index.d.ts",
  "files": ["dist"],
  "scripts": {
    "build": "tsc",
    "dev": "tsc --watch",
    "test": "jest"
  },
  "unicorePlugin": {
    "name": "crm-exporter",
    "displayName": "CRM Exporter",
    "version": "1.0.0",
    "description": "Export contacts to CSV and Google Sheets",
    "minGeekVersion": "1.0.0",
    "author": "My Org <dev@myorg.com>",
    "permissions": [
      "crm:read",
      "files:write"
    ],
    "commands": true,
    "screens": true,
    "replBindings": true
  },
  "peerDependencies": {
    "@unicore-geek/plugin-sdk": "^1.0.0"
  },
  "devDependencies": {
    "@unicore-geek/plugin-sdk": "^1.0.0",
    "typescript": "^5.5.0"
  }
}
```

### Permission Scopes

Declare all permissions your plugin requires. Users are shown a consent prompt on install.

| Permission | Access |
|-----------|--------|
| `crm:read` | Read contacts, companies, deals |
| `crm:write` | Create/update/delete CRM records |
| `erp:read` | Read invoices, products, orders |
| `erp:write` | Create/update ERP records |
| `files:read` | Read uploaded files and RAG documents |
| `files:write` | Upload files and trigger indexing |
| `workflows:read` | Read workflow definitions |
| `workflows:execute` | Trigger workflow runs |
| `agents:chat` | Send messages to AI agents |
| `settings:read` | Read system settings |
| `settings:write` | Modify system settings (requires OWNER role) |

## Plugin Entry Point

The default export must implement the `GeekPlugin` interface:

```typescript
// src/index.ts
import {
  GeekPlugin,
  PluginContext,
  PluginManifest,
} from '@unicore-geek/plugin-sdk';
import { registerCommands } from './commands';
import { registerScreens } from './screens';
import { registerReplBindings } from './repl/bindings';

export default class CrmExporterPlugin implements GeekPlugin {
  readonly manifest: PluginManifest = {
    name: 'crm-exporter',
    version: '1.0.0',
  };

  async onLoad(ctx: PluginContext): Promise<void> {
    ctx.logger.info('CRM Exporter plugin loaded');

    registerCommands(ctx);
    registerScreens(ctx);
    registerReplBindings(ctx);
  }

  async onUnload(ctx: PluginContext): Promise<void> {
    ctx.logger.info('CRM Exporter plugin unloaded');
    // Clean up timers, connections, etc.
  }

  async onError(error: Error, ctx: PluginContext): Promise<void> {
    ctx.logger.error(`Plugin error: ${error.message}`);
  }
}
```

## Plugin Context API

The `PluginContext` object is injected into your plugin at load time and provides access to the full SDK:

```typescript
interface PluginContext {
  // Logging
  logger: Logger;

  // CLI registration
  cli: CliRegistry;

  // TUI registration
  tui: TuiRegistry;

  // REPL registration
  repl: ReplRegistry;

  // UniCore SDK (scoped to declared permissions)
  sdk: UniCoreSdk;

  // Plugin configuration store
  config: PluginConfig;

  // Output helpers
  output: OutputHelper;
}
```

## CLI Commands

Register custom CLI commands via `ctx.cli.register`:

```typescript
// src/commands/index.ts
import { PluginContext } from '@unicore-geek/plugin-sdk';

export function registerCommands(ctx: PluginContext): void {
  // Register: unicore crm-exporter export-csv
  ctx.cli.register('crm-exporter:export-csv', async (args, _ctx) => {
    const output = args['--output'] ?? './contacts.csv';

    ctx.output.spinner('Fetching contacts...');
    const contacts = await ctx.sdk.crm.listContacts({ limit: 10000 });
    ctx.output.stopSpinner();

    const csv = contactsToCsv(contacts);
    await writeFile(output, csv, 'utf-8');

    ctx.output.success(`Exported ${contacts.length} contacts to ${output}`);
  }, {
    description: 'Export all CRM contacts to a CSV file',
    args: {
      '--output': {
        type: 'string',
        alias: '-o',
        description: 'Output file path (default: ./contacts.csv)',
      },
    },
  });

  // Register: unicore crm-exporter stats
  ctx.cli.register('crm-exporter:stats', async (_args, _ctx) => {
    const contacts = await ctx.sdk.crm.listContacts();
    const leads = contacts.filter(c => c.status === 'lead').length;
    const customers = contacts.filter(c => c.status === 'customer').length;

    ctx.output.table([
      { label: 'Total Contacts', value: contacts.length },
      { label: 'Leads', value: leads },
      { label: 'Customers', value: customers },
    ]);
  }, {
    description: 'Show CRM statistics',
  });
}
```

Commands are invoked as:

```bash
unicore crm-exporter:export-csv --output ./my-contacts.csv
unicore crm-exporter:stats
```

## TUI Screens

Register TUI screens (built with [Ink](https://github.com/vadimdemedes/ink), a React renderer for terminals):

```typescript
// src/screens/index.tsx
import React from 'react';
import { Box, Text, useInput } from 'ink';
import { PluginContext, Contact } from '@unicore-geek/plugin-sdk';

export function registerScreens(ctx: PluginContext): void {
  ctx.tui.register('crm-exporter', {
    title: 'CRM Exporter',
    description: 'Export contacts and view CRM stats',
    component: CrmExporterScreen,
    props: { ctx },
  });
}

interface CrmExporterScreenProps {
  ctx: PluginContext;
}

function CrmExporterScreen({ ctx }: CrmExporterScreenProps): JSX.Element {
  const [contacts, setContacts] = React.useState<Contact[]>([]);
  const [loading, setLoading] = React.useState(true);
  const [selectedIndex, setSelectedIndex] = React.useState(0);

  React.useEffect(() => {
    ctx.sdk.crm.listContacts().then(data => {
      setContacts(data);
      setLoading(false);
    });
  }, [ctx]);

  useInput((input, key) => {
    if (key.downArrow) setSelectedIndex(i => Math.min(i + 1, contacts.length - 1));
    if (key.upArrow) setSelectedIndex(i => Math.max(i - 1, 0));
    if (input === 'e') exportCsv(contacts);
  });

  if (loading) {
    return <Text>Loading contacts...</Text>;
  }

  return (
    <Box flexDirection="column">
      <Text bold>CRM Exporter — {contacts.length} contacts</Text>
      <Text dimColor>↑/↓ navigate  e export  q quit</Text>
      {contacts.slice(0, 20).map((c, i) => (
        <Box key={c.id}>
          <Text color={i === selectedIndex ? 'cyan' : undefined}>
            {i === selectedIndex ? '▶ ' : '  '}
            {c.name.padEnd(30)} {c.email}
          </Text>
        </Box>
      ))}
    </Box>
  );
}

async function exportCsv(contacts: Contact[]): Promise<void> {
  // ... export logic
}
```

Launch with:

```bash
unicore tui --plugin crm-exporter
```

## REPL Bindings

Register global bindings available in the REPL session:

```typescript
// src/repl/bindings.ts
import { PluginContext } from '@unicore-geek/plugin-sdk';

export function registerReplBindings(ctx: PluginContext): void {
  ctx.repl.bind('crmExporter', {
    exportCsv: async (outputPath: string) => {
      const contacts = await ctx.sdk.crm.listContacts({ limit: 10000 });
      // ... write CSV
      return `Exported ${contacts.length} contacts to ${outputPath}`;
    },
    stats: async () => {
      const contacts = await ctx.sdk.crm.listContacts();
      return {
        total: contacts.length,
        leads: contacts.filter(c => c.status === 'lead').length,
        customers: contacts.filter(c => c.status === 'customer').length,
      };
    },
  });
}
```

Available in REPL as:

```javascript
> await crmExporter.stats()
{ total: 142, leads: 89, customers: 53 }

> await crmExporter.exportCsv('./contacts.csv')
'Exported 142 contacts to ./contacts.csv'
```

## Plugin Configuration

Plugins can store and retrieve persistent configuration:

```typescript
// Store a value (persisted to ~/.unicore/plugins/crm-exporter/config.json)
await ctx.config.set('defaultExportPath', './exports/');
await ctx.config.set('googleSheetsApiKey', 'AIza...');

// Retrieve a value
const exportPath = await ctx.config.get<string>('defaultExportPath') ?? './';

// Delete a value
await ctx.config.delete('googleSheetsApiKey');

// Get all config
const allConfig = await ctx.config.getAll();
```

## Testing Your Plugin

### Unit Tests (Jest)

```typescript
// test/commands.spec.ts
import { createMockPluginContext } from '@unicore-geek/plugin-sdk/testing';
import { registerCommands } from '../src/commands';

describe('crm-exporter:stats command', () => {
  it('displays contact statistics', async () => {
    const ctx = createMockPluginContext({
      sdk: {
        crm: {
          listContacts: jest.fn().mockResolvedValue([
            { id: '1', name: 'Acme', email: 'a@acme.com', status: 'lead' },
            { id: '2', name: 'Beta', email: 'b@beta.com', status: 'customer' },
          ]),
        },
      },
    });

    registerCommands(ctx);

    await ctx.cli.invoke('crm-exporter:stats', []);

    expect(ctx.output.table).toHaveBeenCalledWith(expect.arrayContaining([
      { label: 'Total Contacts', value: 2 },
      { label: 'Leads', value: 1 },
      { label: 'Customers', value: 1 },
    ]));
  });
});
```

```bash

pnpm build

# Install locally
unicore plugin install ./

# Test commands
unicore crm-exporter:stats
unicore crm-exporter:export-csv --output /tmp/test.csv

# Uninstall
unicore plugin remove @my-org/unicore-plugin-crm-exporter
```

## Publishing Your Plugin

```bash
# Build
pnpm build

# Publish to npm
npm publish --access public

# Users can then install:
unicore plugin install @my-org/unicore-plugin-crm-exporter
```

For private plugins, use a private npm registry (Verdaccio, GitHub Packages, npm private registry).
